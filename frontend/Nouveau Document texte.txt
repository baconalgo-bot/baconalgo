    <script>
        async function scan() {
            document.getElementById('results').innerHTML = '<p>Scanning...</p>';
            
            try {
                const response = await fetch('http://localhost:8000/api/scan');
                const data = await response.json();
                
                let html = '<h2>‚úÖ Found ' + data.signals.length + ' signals!</h2>';
                
                data.signals.forEach(signal => {
                    html += `
                        <div class="signal">
                            <h3>${signal.symbol}</h3>
                            <p>Price: $${signal.price.toFixed(2)}</p>
                            <p>Direction: ${signal.direction}</p>
                            <p>Score: ${signal.score}/100</p>
                        </div>
                    `;
                });
                
                document.getElementById('results').innerHTML = html;
            } catch (error) {
                document.getElementById('results').innerHTML = '<p>Error: ' + error + '</p>';
            }
        }
    </script>
